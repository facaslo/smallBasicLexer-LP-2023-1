Assign -> id tkn_equals LExpr Assign | epsilon,
TExpr -> TValue RestTExpr ,
RestTExpr -> tkn_plus TValue RestTExpr  | epsilon,
TValue -> TextValue | LExpr,
LExpr ->  lValue RestLExpr,
RestLExpr -> Or lValue RestLExpr | And lValue RestLExpr | epsilon,
lValue -> TruthValue | CValue ,
CValue -> Expr RestCvalue,
RestCvalue -> tkn_less Expr | tkn_leq Expr  | tkn_greater Expr | tkn_geq Expr | tkn_diff Expr | epsilon, 
Expr -> Term RestExpr,
RestExpr -> tkn_minus Term RestExpr | epsilon , 
Term -> Factor RestTerm , 
RestTerm -> tkn_times Factor RestTerm | tkn_div Factor RestTerm | epsilon , 
Factor -> tkn_minus RestFactor | RestFactor,
RestFactor -> tkn_number | id | tkn_left_paren TExpr tkn_right_paren,
TruthValue -> True | False ,
TextValue -> tkn_text
